\begin{frame}{Problem Statement}
    Let $A$, $B$, and $C$ be three events such that:
    \begin{equation*}
        P(A) = 0.4, \quad P(B) = 0.5, \quad P(A \cup B) = 0.6,
    \end{equation*}
    \begin{equation*}
        P(C) = 0.6, \quad \text{and} \quad P(A \cap B \cap C^c) = 0.1
    \end{equation*}
    
    Then $P(A \cap B \cap C) =$
    
    \vspace{0.5cm}
    a) $\frac{1}{2}$ \hspace{1cm} b) $\frac{1}{3}$ \hspace{1cm} c) $\frac{1}{4}$ \hspace{1cm} d) $\frac{1}{5}$
\end{frame}

\begin{frame}{Indicator Random Variables}
    To solve this fundamentally, we define indicator random variables for the events $A$, $B$, and $C$. Let $X_A, X_B, X_C \in \{0, 1\}$.
    
    The probability of an event is the expectation of its indicator variable:
    \begin{equation}
        E[X_A] &= P(A) = 0.4 \\
        E[X_B] &= P(B) = 0.5 \\
        E[X_{A \cup B}] &= P(A \cup B) = 0.6 \\
        E[X_A X_B (1 - X_C)] &= P(A \cap B \cap C^c) = 0.1
    \end{equation}
    
    We need to find the expectation of the intersection of all three events:
    \begin{equation}
        E[X_A X_B X_C] = P(A \cap B \cap C)
    \end{equation}
\end{frame}

\begin{frame}{Evaluating the Intersection of A and B}
    From Boolean algebra, the indicator for the union of two events is:
    \begin{equation}
        X_{A \cup B} = X_A + X_B - X_A X_B
    \end{equation}
    
    Taking the expectation on both sides and using the linearity of expectation:
    \begin{equation}
        E[X_{A \cup B}] = E[X_A] + E[X_B] - E[X_A X_B]
    \end{equation}
    
    Substituting the known probabilities:
    \begin{align}
        0.6 &= 0.4 + 0.5 - E[X_A X_B] \\
        0.6 &= 0.9 - E[X_A X_B] \\
        E[X_A X_B] &= 0.3
    \end{align}
    This gives us $P(A \cap B) = 0.3$.
\end{frame}

\begin{frame}{Law of Total Probability via Indicators}
    The intersection $A \cap B$ can be partitioned by event $C$ and its complement $C^c$. In terms of indicator variables:
    \begin{equation}
        X_A X_B = X_A X_B (X_C + (1 - X_C))
    \end{equation}
    \begin{equation}
        X_A X_B = X_A X_B X_C + X_A X_B (1 - X_C)
    \end{equation}
    
    Taking the expectation on both sides:
    \begin{equation}
        E[X_A X_B] = E[X_A X_B X_C] + E[X_A X_B (1 - X_C)]
    \end{equation}
\end{frame}

\begin{frame}{Final Calculation}
    Substitute the known values into the expectation equation:
    \begin{align}
        0.3 &= E[X_A X_B X_C] + 0.1 \\
        E[X_A X_B X_C] &= 0.3 - 0.1 \\
        E[X_A X_B X_C] &= 0.2
    \end{align}
    
    Converting the expectation back to probability notation:
    \begin{equation}
        P(A \cap B \cap C) = 0.2 = \frac{1}{5}
    \end{equation}
    
    \vspace{0.5cm}
    \textbf{Correct Option: (d)}
\end{frame}

import numpy as np
import matplotlib.pyplot as plt

# Number of simulations
N = 100000

# We partition the sample space into disjoint regions based on the problem
# Region 1: A AND B AND C (Expected probability = 0.2)
# Region 2: A AND B AND C_complement (Given probability = 0.1)
# Region 3: A AND not B (P(A) - P(A n B) = 0.4 - 0.3 = 0.1)
# Region 4: B AND not A (P(B) - P(A n B) = 0.5 - 0.3 = 0.2)
# Region 5: Neither A nor B (1 - P(A U B) = 1 - 0.6 = 0.4)

# Generate random variables based on these exact disjoint probabilities
probs = [0.2, 0.1, 0.1, 0.2, 0.4]
events = np.random.choice([1, 2, 3, 4, 5], size=N, p=probs)

# Calculate simulated probability of (A n B n C) which corresponds to Region 1
simulated_prob = np.sum(events == 1) / N

# Visualization
labels = ['A \u2229 B \u2229 C', 'A \u2229 B \u2229 C^c', 'Other Regions']
values = [simulated_prob, np.sum(events == 2)/N, np.sum(np.isin(events, [3,4,5]))/N]

plt.figure(figsize=(7, 5))
plt.bar(labels, values, color=['blue', 'orange', 'gray'])
plt.axhline(y=0.2, color='r', linestyle='--', label='Theoretical P(A \u2229 B \u2229 C) = 0.2')
plt.ylabel('Probability')
plt.title(f'Monte Carlo Simulation (N={N})\nSimulated P(A \u2229 B \u2229 C) = {simulated_prob:.4f}')
plt.legend()
plt.savefig('probability_plot.png')
\end{lstlisting}
